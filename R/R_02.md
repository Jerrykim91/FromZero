## 내장함수 -2 
```r
c()             # combine, 기존 벡터를 결합하여 새로운 벡터를 생성
rep()           # 벡터를 반복시켜 새로운 벡터를 생성
names()         # 벡터의 이름을 확인하거나 수정
length()        # 벡터의 원소 갯수
which()         # 조건에 맞는 벡터의 원소 인덱스를 알려줌

sort()          # 오름차순 정렬
summary()       # 최소값, 최대값, 중간값, 평균값 등을 계산
                # numeric vector만 가능**
var()           # 평균 주변으로 널리 퍼진 데이터를 의미
sd()            # 편차 => 값이 얼마나 다른가를 의미
sum()           # 전달인자의 합 계산
seq() 혹은 1:5  # 일정하게 증가하는 데이터 생성
sample()        # 랜덤수

---

# 데이터 확인
ls()          # 현재 메모리에 올려져있는 변수 목록 확인
head()        # 벡터, 매트릭스, 테이블, 데이터프레임 등의 
              # 1에서 6줄까지 보여줌 => 디폴트 6
tail()        # 뒤에서부터 6개 데이터 보여줌

---

# 설치 
install.package('패키지이름') # 패키지들을 선택하여 다운
# 불러오기
library('패키지이름')         # 패키지 불러오기 => 파이썬의 import 개념

---

# 타입관련 함수 
typeof()       # 데이터형태 확인
class()        # 객체지향 관점에서 데이터형태 확인
is.integer()   # 정수형인지 확인
as.integer()   # 원하는 데이터형으로 형변환

---

# 그래프 
boxplot()     # 변수의 범위와 쏠림을 다른 변수와 비교 표현함
hist()        # numeric vector에 대해 히스토그램을 그림
plot()        # 데이터를 x y 좌표상의 점으로 시각화
prop.table()  # 입력한 항목의 빈도를 %로 계산

---

# help
help() 또는 ? # 함수들의 정보를 표시
help.search() 또는 ?? # 키워드를 쓰는 함수들 찾아줌
example()     # 해당 키워드를 사용하는 예제 찾아줌

```
