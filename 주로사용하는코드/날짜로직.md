```py
def make_date(date): # date

    year = str(date.year)
    # month
    if date.month >= 10:
        month = str(date.month)
    else:
        month = "0" + str(date.month)
    # day
    if date.day >= 10 :
        day = str(date.day)
    else:
        day = "0" + str(date.day)
    
    # hour
    hour = str(date.hour)

    f_date = year + "." + month + "." + day + "." + hour + ":00"
    return f_date


```
